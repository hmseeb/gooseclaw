1. **Add a goose web health monitor thread** that runs in the background:
   ```python
   def goose_health_monitor():
       """Monitor goose web subprocess and auto-restart on crash with backoff."""
       backoff = 5  # initial backoff seconds
       max_backoff = 120
       consecutive_failures = 0

       while True:
           time.sleep(15)  # check every 15 seconds
           if not is_configured():
               continue

           with goose_lock:
               proc = goose_process
           if proc is None:
               continue

           if proc.poll() is not None:
               # process has exited
               exit_code = proc.returncode
               consecutive_failures += 1
               wait_time = min(backoff * (2 ** (consecutive_failures - 1)), max_backoff)
               print(f"[health] goose web exited (code {exit_code}). "
                     f"Restart #{consecutive_failures} in {wait_time}s...")
               time.sleep(wait_time)
               try:
                   start_goose_web()
                   print(f"[health] goose web restarted after failure #{consecutive_failures}")
               except Exception as e:
                   print(f"[health] restart failed: {e}")
           else:
               # process is running, reset backoff on sustained health
               if consecutive_failures > 0:
                   # verify it's actually responding
                   try:
                       conn = http.client.HTTPConnection("127.0.0.1", GOOSE_WEB_PORT, timeout=3)
                       conn.request("GET", "/api/health")
                       resp = conn.getresponse()
                       conn.close()
                       if resp.status == 200:
                           consecutive_failures = 0
                   except Exception:
                       pass  # will catch on next cycle if it dies
   ```

1. **Make save_setup() use atomic write-then-rename:**
   ```python
   def save_setup(config):
       """Atomically write config to setup.json (write tmp, then rename)."""
       os.makedirs(CONFIG_DIR, exist_ok=True)
       tmp_path = SETUP_FILE + ".tmp"
       try:
           with open(tmp_path, "w") as f:
               json.dump(config, f, indent=2)
           os.chmod(tmp_path, 0o600)
           os.replace(tmp_path, SETUP_FILE)  # atomic on same filesystem
       except Exception:
           # clean up temp file on failure
           try:
               os.unlink(tmp_path)
           except OSError:
               pass
           raise
   ```
   `os.replace()` is atomic on POSIX when source and dest are on the same filesystem (which they are — both in CONFIG_DIR).

2. **Add config backup before overwrite.** Before save_setup writes, copy existing setup.json to setup.json.bak:
   ```python
   # in save_setup, before writing:
   if os.path.exists(SETUP_FILE):
       try:
           import shutil
           shutil.copy2(SETUP_FILE, SETUP_FILE + ".bak")
       except Exception:
           pass  # non-fatal
   ```

3. **Reduce proxy timeout** in `proxy_to_goose()`. Currently `timeout=300` (5 minutes!). Change to:
   ```python
   PROXY_TIMEOUT = int(os.environ.get("GOOSECLAW_PROXY_TIMEOUT", "60"))
   ```
   And use it in the HTTPConnection: `conn = http.client.HTTPConnection("127.0.0.1", GOOSE_WEB_PORT, timeout=PROXY_TIMEOUT)`

   BUT for SSE (text/event-stream) connections, use a longer timeout since they are long-lived:
   - First, make the initial connection with PROXY_TIMEOUT
   - For SSE streaming, don't set socket timeout (it's a long-lived connection)
   - For non-SSE, PROXY_TIMEOUT is fine

4. **Improve graceful shutdown** in the signal handler:
   ```python
   def shutdown(_sig, _frame):
       print("[gateway] shutting down...")
       # stop accepting new connections
       threading.Thread(target=server.shutdown, daemon=True).start()
       # terminate child processes with timeout
       stop_goose_web()
       with telegram_lock:
           if telegram_process and telegram_process.poll() is None:
               telegram_process.terminate()
               try:
                   telegram_process.wait(timeout=5)
               except subprocess.TimeoutExpired:
                   telegram_process.kill()
       print("[gateway] shutdown complete")
   ```
   Key change: stop accepting connections FIRST, then clean up children. Add wait+kill pattern for telegram (currently just terminate with no wait).

5. **Fix entrypoint.sh signal propagation** (item 37). The current trap handler kills processes but uses `||` which masks failures. Improve:
   ```bash
   shutdown() {
       echo "[shutdown] SIGTERM received"
       if [ -n "${GITHUB_PAT:-}" ] && [ -n "${GITHUB_REPO:-}" ]; then
           echo "[shutdown] persisting final state..."
           "$APP_DIR/scripts/persist.sh" 2>&1 || true
       fi
       # send SIGTERM to gateway (it handles its own children)
       if kill -0 "$GATEWAY_PY_PID" 2>/dev/null; then
           kill -TERM "$GATEWAY_PY_PID"
           wait "$GATEWAY_PY_PID" 2>/dev/null || true
       fi
       [ -n "${PERSIST_PID:-}" ] && kill -TERM "$PERSIST_PID" 2>/dev/null || true
       [ -n "${WATCHDOG_PID:-}" ] && kill -TERM "$WATCHDOG_PID" 2>/dev/null || true
       echo "[shutdown] done"
       exit 0
   }
   ```
   Key change: send SIGTERM to gateway and WAIT for it (so it can clean up children), instead of fire-and-forget kill.
