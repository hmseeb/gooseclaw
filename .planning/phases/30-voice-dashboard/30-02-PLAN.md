**1. AudioWorklet Processor Code (as string constant):**

Define `WORKLET_CODE` as a template literal string containing a complete AudioWorkletProcessor:

```javascript
const WORKLET_CODE = `
class PCMCaptureProcessor extends AudioWorkletProcessor {
  constructor() {
    super();
    this.buffer = new Int16Array(2048);
    this.writeIndex = 0;
  }
  process(inputs) {
    const input = inputs[0];
    if (!input || !input[0]) return true;
    const samples = input[0];
    for (let i = 0; i < samples.length; i++) {
      const s = Math.max(-1, Math.min(1, samples[i]));
      this.buffer[this.writeIndex++] = s * 0x7FFF;
      if (this.writeIndex >= this.buffer.length) {
        this.port.postMessage(this.buffer.buffer.slice(0));
        this.writeIndex = 0;
      }
    }
    return true;
  }
}
registerProcessor('pcm-capture', PCMCaptureProcessor);
`;
```

**1. AudioPlayback class:**

```javascript
class AudioPlayback {
  constructor(ctx) {
    this.ctx = ctx;
    this.queue = [];
    this.scheduledTime = 0;
    this.playing = false;
    this.scheduledSources = [];
    this.SCHEDULE_AHEAD = 0.2;
    this.INITIAL_BUFFER = 0.1;
    this._rafId = null;
  }

  addPCM16(chunk) {
    // chunk is Uint8Array of Int16 LE PCM at 24kHz
    const float32 = new Float32Array(chunk.length / 2);
    const view = new DataView(chunk.buffer, chunk.byteOffset, chunk.byteLength);
    for (let i = 0; i < float32.length; i++) {
      float32[i] = view.getInt16(i * 2, true) / 32768;
    }
    this.queue.push(float32);
    if (!this.playing) {
      this.playing = true;
      this.scheduledTime = this.ctx.currentTime + this.INITIAL_BUFFER;
      this._scheduleNext();
    }
  }

  _scheduleNext() {
    while (this.queue.length > 0 &&
           this.scheduledTime < this.ctx.currentTime + this.SCHEDULE_AHEAD) {
      const samples = this.queue.shift();
      const buffer = this.ctx.createBuffer(1, samples.length, 24000);
      buffer.getChannelData(0).set(samples);
      const source = this.ctx.createBufferSource();
      source.buffer = buffer;
      source.connect(this.analyserNode || this.ctx.destination);
      source.onended = () => {
        const idx = this.scheduledSources.indexOf(source);
        if (idx !== -1) this.scheduledSources.splice(idx, 1);
        // If no more queued audio and no more scheduled sources, AI done speaking
        if (this.queue.length === 0 && this.scheduledSources.length === 0 && this.playing) {
          this.playing = false;
          if (currentState === STATE.SPEAKING) {
            setState(STATE.READY);
          }
        }
      };
      source.start(this.scheduledTime);
      this.scheduledSources.push(source);
      this.scheduledTime += samples.length / 24000;
    }
    if (this.queue.length > 0 || this.scheduledSources.length > 0) {
      this._rafId = requestAnimationFrame(() => this._scheduleNext());
    }
  }

  clearQueue() {
    this.queue = [];
    // Stop all scheduled sources immediately
    for (const source of this.scheduledSources) {
      try { source.stop(); } catch(e) { /* may already be stopped */ }
    }
    this.scheduledSources = [];
    this.playing = false;
    this.scheduledTime = 0;
    if (this._rafId) {
      cancelAnimationFrame(this._rafId);
      this._rafId = null;
    }
  }

  setAnalyser(node) {
    this.analyserNode = node;
  }
}
```

Key design decisions:
- `source.connect(this.analyserNode || this.ctx.destination)` -- routes through analyser if set (Plan 03 hooks this up for visualizer), otherwise direct to speakers.
- `source.onended` tracks when all audio finishes to transition from SPEAKING -> READY.
- `clearQueue()` immediately stops all scheduled sources (barge-in requires instant silence).
- `createBuffer(1, samples.length, 24000)` specifies 24kHz sample rate regardless of AudioContext rate (browser handles resampling automatically).
