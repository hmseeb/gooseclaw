2. **Visualizer drawing function:**

   ```javascript
   function startVisualizer() {
     const canvas = document.getElementById('visualizer-canvas');
     if (!canvas) return;
     const ctx = canvas.getContext('2d');
     const dpr = window.devicePixelRatio || 1;
     canvas.width = canvas.clientWidth * dpr;
     canvas.height = canvas.clientHeight * dpr;
     ctx.scale(dpr, dpr);
     const w = canvas.clientWidth;
     const h = canvas.clientHeight;
     const cx = w / 2;
     const cy = h / 2;
     const dataArray = analyserNode ? new Uint8Array(analyserNode.frequencyBinCount) : null;

     function draw() {
       visualizerAnimId = requestAnimationFrame(draw);
       ctx.clearRect(0, 0, w, h);

       let amplitude = 0;
       if (analyserNode && dataArray) {
         analyserNode.getByteFrequencyData(dataArray);
         let sum = 0;
         for (let i = 0; i < dataArray.length; i++) sum += dataArray[i];
         amplitude = sum / dataArray.length / 255; // 0 to 1
       }

       // Base radius + amplitude-driven expansion
       const baseRadius = Math.min(w, h) * 0.15;
       const maxExpand = Math.min(w, h) * 0.2;
       const radius = baseRadius + amplitude * maxExpand;

       // Color based on current state
       let r, g, b, alpha;
       switch (currentState) {
         case STATE.DISCONNECTED:
           r = 80; g = 80; b = 80; alpha = 0.3;
           break;
         case STATE.CONNECTING:
           r = 100; g = 100; b = 100; alpha = 0.5;
           break;
         case STATE.READY:
           r = 124; g = 106; b = 239; alpha = 0.5;
           break;
         case STATE.LISTENING:
           r = 239; g = 68; b = 68; alpha = 0.4 + amplitude * 0.6;
           break;
         case STATE.THINKING:
           r = 245; g = 158; b = 11; alpha = 0.5;
           break;
         case STATE.SPEAKING:
           r = 34; g = 197; b = 94; alpha = 0.4 + amplitude * 0.6;
           break;
         default:
           r = 80; g = 80; b = 80; alpha = 0.3;
       }

       // Outer glow
       const gradient = ctx.createRadialGradient(cx, cy, radius * 0.5, cx, cy, radius * 1.5);
       gradient.addColorStop(0, `rgba(${r}, ${g}, ${b}, ${alpha * 0.3})`);
       gradient.addColorStop(1, `rgba(${r}, ${g}, ${b}, 0)`);
       ctx.beginPath();
       ctx.arc(cx, cy, radius * 1.5, 0, Math.PI * 2);
       ctx.fillStyle = gradient;
       ctx.fill();

       // Main orb
       ctx.beginPath();
       ctx.arc(cx, cy, radius, 0, Math.PI * 2);
       ctx.fillStyle = `rgba(${r}, ${g}, ${b}, ${alpha})`;
       ctx.fill();

       // Inner highlight
       const inner = ctx.createRadialGradient(cx - radius * 0.2, cy - radius * 0.2, 0, cx, cy, radius);
       inner.addColorStop(0, `rgba(255, 255, 255, ${alpha * 0.15})`);
       inner.addColorStop(1, `rgba(255, 255, 255, 0)`);
       ctx.beginPath();
       ctx.arc(cx, cy, radius, 0, Math.PI * 2);
       ctx.fillStyle = inner;
       ctx.fill();
     }

     draw();
   }
   ```
