**3. Modify `_voice_relay_gemini_to_browser` (line 9095):**

For `parsed["type"] == "tool_call"`:
- Extract `tool_data = parsed["data"]`
- For each `fc` in `tool_data.get("functionCalls", [])`:
  - Get `call_id = fc.get("id", "")`, `call_name = fc.get("name", "")`, `call_args = fc.get("args", {})`
  - Look up original name: `original_name = session_state.get("tool_name_map", {}).get(call_name, call_name)`
  - Send "running" status to browser: `ws_send_frame(browser_sock, WS_OP_TEXT, json.dumps({"type": "tool_status", "name": call_name, "status": "running"}).encode())`
  - Get tool_session_id from session_state
  - Spawn a daemon thread to execute the tool:
    ```python
    def _exec_tool(cid, cname, cargs, oname, sid):
        try:
            result = _voice_execute_tool(cname, cargs, sid, oname)
            # Build and send toolResponse to Gemini
            response_msg = _voice_build_tool_response(cid, cname, result)
            with session_state["_lock"]:
                gs = session_state["gemini_sock"]
                if gs:
                    ws_send_frame(gs, WS_OP_TEXT,
                        json.dumps(response_msg).encode(), mask=True)
            # Notify browser: done
            summary = result.get("result", result.get("error", ""))[:200]
            status = "done" if "result" in result else "error"
            ws_send_frame(browser_sock, WS_OP_TEXT,
                json.dumps({"type": "tool_status", "name": cname,
                            "status": status, "result": summary}).encode())
        except Exception as e:
            _voice_log.error(f"Tool exec error: {e}")
            # ALWAYS send toolResponse even on error (prevents Gemini hang)
            err_response = _voice_build_tool_response(cid, cname, {"error": str(e)})
            try:
                with session_state["_lock"]:
                    gs = session_state["gemini_sock"]
                    if gs:
                        ws_send_frame(gs, WS_OP_TEXT,
                            json.dumps(err_response).encode(), mask=True)
            except Exception:
                pass
            ws_send_frame(browser_sock, WS_OP_TEXT,
                json.dumps({"type": "tool_status", "name": cname,
                            "status": "error", "result": str(e)[:200]}).encode())

    t = threading.Thread(
        target=_exec_tool,
        args=(call_id, call_name, call_args, original_name,
              session_state.get("tool_session_id")),
        daemon=True)
    t.start()
    ```
  - IMPORTANT: Do NOT block the relay thread waiting for tool execution

**2. JavaScript — add `addToolStatus` function (near the existing `addTranscript` function):**
```javascript
function addToolStatus(name, status, result) {
    var el = document.getElementById('tool-' + name.replace(/[^a-zA-Z0-9]/g, '-'));
    if (!el) {
        el = document.createElement('div');
        el.id = 'tool-' + name.replace(/[^a-zA-Z0-9]/g, '-');
        document.getElementById('transcript').appendChild(el);
    }
    el.className = 'tool-status' + (status === 'error' ? ' error' : '') +
                   (status === 'cancelled' ? ' cancelled' : '');

    var stateText = status === 'running' ? 'running' :
                    status === 'done' ? 'done' :
                    status === 'error' ? 'failed' : 'cancelled';

    el.innerHTML = '<div class="tool-badge">' +
        '<span class="tool-name">' + escapeHtml(name) + '</span>' +
        '<span class="tool-state ' + status + '">' + stateText + '</span>' +
        '</div>' +
        (result ? '<div class="tool-result">' + escapeHtml(result) + '</div>' : '');

    autoScrollTranscript();
}
```
