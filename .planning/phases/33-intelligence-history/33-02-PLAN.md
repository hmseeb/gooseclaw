```javascript
function toggleHistory() {
  var panel = document.getElementById('history-panel');
  if (panel.classList.contains('active')) {
    panel.classList.remove('active');
  } else {
    panel.classList.add('active');
    loadHistory();
  }
}

async function loadHistory() {
  var list = document.getElementById('history-list');
  var detail = document.getElementById('session-detail');
  list.classList.remove('hidden');
  detail.classList.remove('active');
  list.innerHTML = '<div class="history-empty">Loading...</div>';
  try {
    var resp = await fetch('/api/voice/sessions');
    if (!resp.ok) throw new Error('Failed to load');
    var data = await resp.json();
    list.innerHTML = '';
    if (!data.sessions || data.sessions.length === 0) {
      list.innerHTML = '<div class="history-empty">No voice sessions yet</div>';
      return;
    }
    data.sessions.forEach(function(s) {
      var el = document.createElement('div');
      el.className = 'history-item';
      var d = new Date(s.started_at);
      var dur = s.duration_seconds ? Math.round(s.duration_seconds / 60) + ' min' : '';
      el.innerHTML = '<div class="history-date">' + d.toLocaleDateString() + ' ' + d.toLocaleTimeString() + '</div>' +
                     '<div class="history-preview">' + escapeHtml(s.preview || 'No preview') + '</div>' +
                     '<div class="history-meta">' + escapeHtml(s.voice_name || 'Aoede') + (dur ? ' · ' + dur : '') +
                     (s.memory_extracted ? ' · memorized' : '') + '</div>';
      el.onclick = function() { viewSession(s.id); };
      list.appendChild(el);
    });
  } catch(e) {
    list.innerHTML = '<div class="history-empty">Failed to load history</div>';
  }
}

async function viewSession(id) {
  var list = document.getElementById('history-list');
  var detail = document.getElementById('session-detail');
  var transcript = document.getElementById('session-transcript');
  var title = document.getElementById('session-detail-title');
  list.classList.add('hidden');
  detail.classList.add('active');
  transcript.innerHTML = '<div class="history-empty">Loading...</div>';
  try {
    var resp = await fetch('/api/voice/sessions/' + id);
    if (!resp.ok) throw new Error('Not found');
    var data = await resp.json();
    var d = new Date(data.started_at);
    title.textContent = d.toLocaleDateString() + ' ' + d.toLocaleTimeString();
    transcript.innerHTML = '';
    (data.transcript || []).forEach(function(entry) {
      var el = document.createElement('div');
      if (entry.speaker === 'tool') {
        el.className = 'session-msg tool';
        el.textContent = '🔧 ' + (entry.name || 'tool') + ' — ' + (entry.status || '');
      } else {
        el.className = 'session-msg ' + entry.speaker;
        el.innerHTML = '<div class="session-msg-speaker">' + (entry.speaker === 'user' ? 'You' : 'AI') + '</div>' +
                       escapeHtml(entry.text || '');
      }
      transcript.appendChild(el);
    });
    if (!data.transcript || data.transcript.length === 0) {
      transcript.innerHTML = '<div class="history-empty">No transcript</div>';
    }
  } catch(e) {
    transcript.innerHTML = '<div class="history-empty">Session not found</div>';
  }
}

function backToHistory() {
  document.getElementById('history-list').classList.remove('hidden');
  document.getElementById('session-detail').classList.remove('active');
}
```

```javascript
var selectedVoice = 'Aoede';

function toggleVoicePicker() {
  var picker = document.getElementById('voice-picker');
  if (picker.classList.contains('active')) {
    picker.classList.remove('active');
  } else {
    picker.classList.add('active');
    loadVoices();
  }
}

async function loadVoices() {
  var grid = document.getElementById('voice-grid');
  grid.innerHTML = '';
  try {
    var resp = await fetch('/api/voice/preference');
    if (!resp.ok) throw new Error('Failed');
    var data = await resp.json();
    selectedVoice = data.voice_name || 'Aoede';
    (data.voices || []).forEach(function(v) {
      var el = document.createElement('div');
      el.className = 'voice-option' + (v.name === selectedVoice ? ' selected' : '');
      el.innerHTML = '<div class="voice-option-name">' + escapeHtml(v.name) + '</div>' +
                     '<div class="voice-option-style">' + escapeHtml(v.style) + '</div>';
      el.onclick = function() { selectVoice(v.name); };
      grid.appendChild(el);
    });
  } catch(e) {
    grid.innerHTML = '<div class="history-empty">Failed to load voices</div>';
  }
}

async function selectVoice(name) {
  try {
    var resp = await fetch('/api/voice/preference', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({voice_name: name})
    });
    if (resp.ok) {
      selectedVoice = name;
      // Update UI selection
      var options = document.querySelectorAll('.voice-option');
      options.forEach(function(el) {
        el.classList.toggle('selected', el.querySelector('.voice-option-name').textContent === name);
      });
    }
  } catch(e) {
    // silently fail
  }
}
```
